class Player (manage players)
  - initialize/create players
  - store lives(init: 3)
  - count lives
  - check dead or alive

class Game (manage gameplay)
  - initialize the game
  - get players
  - run the game
    - create a question,
    - check the answer,
    - correct the answer
  - loop gameplay
  - print summary/message

class Question
  - deal questions


// store players in an array so that it will be reusable for more users


class TurnManager (manager to switch the players' roll)
  - identify the current player
  - turn the players

class TurnManager
  def initialize(players)
    @players = players.dup.shuffle
  end

  def current_player
    @players.first
  end

  def next_turn
    @players.rotate!
  end
end
